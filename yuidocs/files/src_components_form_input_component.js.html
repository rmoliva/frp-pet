/** @class components/form/input/component */

'use strict';

const R = require('ramda');
const baseComponent = require('../../base');

const initialState = {
  disabled: false,
  error: false,
  fluid: false,
  icon: '',
  loading: false,
  size: '',
  text: '',
  transparent: false,
};

// TODO
const isMaybe = R.T;

/**
 * This are the actions supported by the component
 */
const actionTypes = {
  /**
   * Set the text of the input field
   * @memberof! actionTypes
   * @method actions.setText
   */
  setText: [isMaybe],
  /**
   * @name actions.setIcon
   * @function
   */
  setIcon: [isMaybe],
  /**
   * @name actions.setSize
   * @function
   */
  setSize: [isMaybe],
  /**
   * @name actions.setLoading
   * @function
   */
  setLoading: [Boolean],
  /**
   * @name actions.setDisabled
   * @function
   */
  setDisabled: [Boolean],
  /**
   * @name actions.setError
   * @function
   */
  setError: [Boolean],
  /**
   * @name actions.setFluid
   * @function
   */
  setFluid: [Boolean],
  /**
   * @name actions.setTransparent
   * @function
   */
  setTransparent: [Boolean],
};

/**
 * @name components/form/input/component
 * @function
 * @param {Object} config - Initial component configuration
 * @param {String} config.text - Initial input text
 * @param {Boolean} config.disabled - Initial disabled state
 * @return {Component} The component object
 */
module.exports = baseComponent({
  initialState: initialState,

  actionTypes: actionTypes,
  actionCase: function(state) {
    return {
      // setText:: (Maybe) => state
      setText: (text) => R.merge(state, {text: text.orSome('')}),
      // setIcon:: (Maybe) => state
      setIcon: (icon) => R.merge(state, {icon: icon.orSome('')}),
      // setSize:: (Maybe) => state
      setSize: (size) => R.merge(state, {size: size.orSome('')}),
      // setLoading:: (boolean) => state
      setLoading: (loading) => R.merge(state, {loading: loading}),
      // setDisabled:: (boolean) => state
      setDisabled: (disabled) => R.merge(state, {disabled: disabled}),
      // setError:: (boolean) => state
      setError: (error) => R.merge(state, {error: error}),
      // setFluid:: (boolean) => state
      setFluid: (fluid) => R.merge(state, {fluid: fluid}),
      // setTransparent:: (boolean) => state
      setTransparent: (transparent) =>
        R.merge(state, {transparent: transparent}),
    };
  },
});
